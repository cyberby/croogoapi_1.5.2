160: /**
161:  * acl and auth
162:  *
163:  * @return void
164:  */
165:     public function auth() {
166:         $this->_configure();
167:         $user = $this->_controller->Auth->user();
168:         // Admin role is allowed to perform all actions, bypassing ACL
169:         if (!empty($user['role_id']) && $user['role_id'] == 1) {
170:             $this->_controller->Auth->allow();
171:             return;
172:         }
173: 
174:         // authorization for authenticated user is handled by authorize object
175:         if ($user) {
176:             return;
177:         }
178: 
179:         // public access authorization
180:         $cacheName = 'permissions_public';
181:         if (($perms = Cache::read($cacheName, 'permissions')) === false) {
182:             $perms = $this->getPermissions('Role', 3);
183:             Cache::write($cacheName, $perms, 'permissions');
184:         }
185:         if (!empty($perms['allowed'][$this->_controller->name])) {
186:             $this->_controller->Auth->allow(
187:                 $perms['allowed'][$this->_controller->name]
188:             );
189:         }
190:     }

192: /**
193:  * getPermissions
194:  * retrieve list of permissions from database
195:  * @param string $model model name
196:  * @param string $id model id
197:  * @return array list of authorized and allowed actions
198:  */
199:     public function getPermissions($model, $id) {
200:         $Acl =& $this->_controller->Acl;
201:         $aro = array('model' => $model, 'foreign_key' => $id);
202:         $node = $Acl->Aro->node($aro);
203:         $nodes = $Acl->Aro->getPath($node[0]['Aro']['id']);
204: 
205:         $aros = Hash::extract($node, '{n}.Aro.id');
206:         if (!empty($nodes)) {
207:             $aros = Hash::merge($aros, Hash::extract($nodes, '{n}.Aro.id'));
208:         }
209: 
210:         $permissions = $Acl->Aro->Permission->find('all', array(
211:             'conditions' => array(
212:                 'Permission.aro_id' => $aros,
213:                 'Permission._create' => 1,
214:                 'Permission._read' => 1,
215:                 'Permission._update' => 1,
216:                 'Permission._delete' => 1,
217:             )
218:         ));
219: 
220:         $authorized = $allowedActions = array();
221:         foreach ($permissions as $permission) {
222:             $path = $Acl->Aco->getPath($permission['Permission']['aco_id']);
223:             if (empty($path)) {
224:                 continue;
225:             }
226:             $acos = count($path);
227:             if ($acos == 4) {
228:                 // plugin controller/action
229:                 $controller = $path[2]['Aco']['alias'];
230:                 $action = $path[3]['Aco']['alias'];
231:             } else if ($acos == 3) {
232:                 // app controller/action
233:                 $controller = $path[1]['Aco']['alias'];
234:                 $action = $path[2]['Aco']['alias'];
235:             } else {
236:                 $this->log(sprintf(
237:                     'Incomplete path for aco_id = %s:',
238:                     $permission['Permission']['id']
239:                 ));
240:                 $this->log($path);
241:             }
242:             $allowedActions[$controller][] = $action;
243:             $authorized[] = implode('/', Hash::extract($path, '{n}.Aco.alias'));
244:         }
245:         return array('authorized' => $authorized, 'allowed' => $allowedActions);
246:     }
