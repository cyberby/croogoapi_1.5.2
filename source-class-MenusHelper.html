200:     public function linkStringToArray($link) {
201:         if (is_array($link)) {
202:             $link = key($link);
203:         }
204:         $link = explode('/', $link);
205:         $linkArr = array();
206:         foreach ($link as $linkElement) {
207:             if ($linkElement != null) {
208:                 $linkElementE = explode(':', $linkElement);
209:                 if (isset($linkElementE['1'])) {
210:                     $linkArr[$linkElementE['0']] = $linkElementE['1'];
211:                 } else {
212:                     $linkArr[] = $linkElement;
213:                 }
214:             }
215:         }
216:         if (!isset($linkArr['plugin'])) {
217:             $linkArr['plugin'] = false;
218:         }
219: 
220:         return $linkArr;

233:             'controller' => false, 'action' => false
234:             ),
235:             $url

237:         foreach ($actions as $key => $val) {
238:             if (is_string($key)) {
239:                 if (is_bool($val)) {
240:                     if ($val === true) {
241:                         $result[] = $key;
242:                     }
243:                 } else {
244:                     $result[] = $key . ':' . $val;
245:                 }
246:             } else {
247:                 $result[] = $val;
248:             }
249:         }
