210: /**
211:  * Activate/move Wysiwyg before Ckeditor/Tinymce when appropriate
212:  */
213:     protected function _bootstrapSetupEditor($plugins) {
214:         $plugins = array_flip($plugins);
215:         if (empty($plugins['Ckeditor']) && empty($plugins['Tinymce'])) {
216:             return;
217:         }
218:         foreach ($plugins as $plugin => &$value) {
219:             $value *= 10;
220:         }
221: 
222:         if (!empty($plugins['Ckeditor']) && !empty($plugins['Tinymce'])) {
223:             $editor = ($plugins['Ckeditor'] < $plugins['Tinymce']) ? $plugins['Ckeditor'] : $plugins['Tinymce'];
224:         } else if (!empty($plugins['Ckeditor'])) {
225:             $editor = $plugins['Ckeditor'];
226:         } else {
227:             $editor = $plugins['Tinymce'];
228:         }
229: 
230:         if (empty($plugins['Wysiwyg'])) {
231:             $plugins['Wysiwyg'] = $editor - 1;
232:         } else {
233:             if ($plugins['Wysiwyg'] >= $editor) {
234:                 $plugins['Wysiwyg'] = $editor - 1;
235:             }
236:         }
237: 
238:         asort($plugins);
239:         $plugins = array_flip($plugins);
240:         return $plugins;
241:     }

243: /**
244:  * Re-order plugins based on dependencies:
245:  * for e.g, Ckeditor depends on Wysiwyg
246:  * if in Hook.bootstraps Ckeditor appears before Wysiwyg,
247:  * we will reorder it so that it loads right after Wysiwyg
248:  */
249:     protected function _bootstrapReorderByDependency($plugins) {
250:         $pluginsOrdered = $plugins;
251:         foreach ($plugins as $p) {
252:             $jsonPath = APP . 'Plugin' . DS . $p . DS . 'Config' . DS . 'plugin.json';
253:             if (file_exists($jsonPath)) {
254:                 $pluginData = json_decode(file_get_contents($jsonPath), true);
255:                 if (isset($pluginData['dependencies']['plugins'])) {
256:                     foreach ($pluginData['dependencies']['plugins'] as $d) {
257:                         $k = array_search($p, $pluginsOrdered);
258:                         $dk = array_search($d, $pluginsOrdered);
259:                         if ($dk > $k) {
260:                             unset($pluginsOrdered[$k]);
261:                             $pluginsOrdered = array_slice($pluginsOrdered, 0, $k + 1, true) +
262:                                 array($p => $p) +
263:                                 array_slice($pluginsOrdered, $k + 1, count($pluginsOrdered) - 1, true);
264:                             $pluginsOrdered = array_values($pluginsOrdered);
265:                         }
266:                     }
267:                 }
268:             }
269:         }
270:         return $pluginsOrdered;
271:     }

276:     protected function _createFirstMigration($plugin) {
277:         static $Migration;

284:         $className = 'FirstMigration' . $plugin;
285:         $migration = $Migration->findByClass($className);
