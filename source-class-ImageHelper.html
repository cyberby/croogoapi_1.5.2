20:      * @param integer $height Height of returned image
21:      * @param boolean $aspect Maintain aspect ratio (default: true)
22:      * @param array $htmlAttributes Array of HTML attributes.
23:      * @param boolean $return Wheter this method should return a value or output it. This overrides AUTO_OUTPUT.

25:      * @access public

27:     public function resize($path, $width, $height, $aspect = true, $htmlAttributes = array(), $return = false) {
28:         $types = array(1 => "gif", "jpeg", "png", "swf", "psd", "wbmp"); // used to determine image type
29:         if (empty($htmlAttributes['alt'])) $htmlAttributes['alt'] = 'thumb';  // Ponemos alt default
30: 
31:         $uploadsDir = 'uploads';
32: 
33:         $fullpath = ROOT.DS.APP_DIR.DS.WEBROOT_DIR.DS.$uploadsDir.DS;
34:         $url = ROOT.DS.APP_DIR.DS.WEBROOT_DIR.DS.$path;
35: 
36:         if (!($size = getimagesize($url)))
37:             return; // image doesn't exist
38: 
39:         if ($aspect) { // adjust to aspect.
40:             if (($size[1]/$height) > ($size[0]/$width))  // $size[0]:width, [1]:height, [2]:type
41:                 $width = ceil(($size[0]/$size[1]) * $height);
42:             else
43:                 $height = ceil($width / ($size[0]/$size[1]));
44:         }
45: 
46:         $relfile = $this->webroot.$uploadsDir.'/'.$this->cacheDir.'/'.$width.'x'.$height.'_'.basename($path); // relative file
47:         $cachefile = $fullpath.$this->cacheDir.DS.$width.'x'.$height.'_'.basename($path);  // location on server
48: 
49:         if (file_exists($cachefile)) {
50:             $csize = getimagesize($cachefile);
51:             $cached = ($csize[0] == $width && $csize[1] == $height); // image is cached
52:             if (@filemtime($cachefile) < @filemtime($url)) // check if up to date
53:                 $cached = false;
54:         } else {
55:             $cached = false;
56:         }
57: 
58:         if (!$cached) {
59:             $resize = ($size[0] > $width || $size[1] > $height) || ($size[0] < $width || $size[1] < $height);
60:         } else {
61:             $resize = false;
62:         }
63: 
64:         if ($resize) {
65:             $image = call_user_func('imagecreatefrom'.$types[$size[2]], $url);
66:             if (function_exists("imagecreatetruecolor") && ($temp = imagecreatetruecolor ($width, $height))) {
67:                 imagecopyresampled ($temp, $image, 0, 0, 0, 0, $width, $height, $size[0], $size[1]);
68:             } else {

71:             }
