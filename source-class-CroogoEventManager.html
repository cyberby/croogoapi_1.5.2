 34: /**
 35:  * Load Event Handlers during bootstrap.
 36:  *
 37:  * Plugins can add their own custom EventHandler in Config/events.php
 38:  * with the following format:
 39:  *
 40:  * $config = array(
 41:  *     'EventHandlers' => array(
 42:  *         'Example.ExampleEventHandler' => array(
 43:  *             'eventKey' => null,
 44:  *             'options' => array(
 45:  *                 'priority' => 1,
 46:  *                 'passParams' => false,
 47:  *      )));
 48:  *
 49:  * @return void
 50:  */
 51:     public static function loadListeners() {
 52:         $eventManager = CroogoEventManager::instance();
 53:         $cached = Cache::read('EventHandlers', 'cached_settings');
 54:         if ($cached === false) {
 55:             $eventHandlers = Configure::read('EventHandlers');
 56:             $validKeys = array('eventKey' => null, 'options' => array());
 57:             $cached = array();
 58:             if (!empty($eventHandlers) && is_array($eventHandlers)) {
 59:                 foreach ($eventHandlers as $eventHandler => $eventOptions) {
 60:                     $eventKey = null;
 61:                     if (is_numeric($eventHandler)) {
 62:                         $eventHandler = $eventOptions;
 63:                         $eventOptions = array();
 64:                     }
 65:                     list($plugin, $class) = pluginSplit($eventHandler);
 66:                     if (!empty($eventOptions)) {
 67:                         extract(array_intersect_key($validKeys, $eventOptions));
 68:                     }
 69:                     App::uses($class, $plugin . '.Event');
 70:                     if (class_exists($class)) {
 71:                         $cached[] = compact('plugin', 'class', 'eventKey', 'eventOptions');
 72:                     } else {
 73:                         CakeLog::error(__d('croogo', 'EventHandler %s not found in plugin %s', $class, $plugin));
 74:                     }
 75:                 }
 76:                 Cache::write('EventHandlers', $cached, 'cached_settings');
 77:             }
 78:         }
 79:         foreach ($cached as $cache) {
 80:             extract($cache);
 81:             if (CakePlugin::loaded($plugin)) {
 82:                 App::uses($class, $plugin . '.Event');
 83:                 $listener = new $class();
 84:                 $eventManager->attach($listener, $eventKey, $eventOptions);
 85:             }
 86:         }
 87:     }

 94:     public function attach($callable, $eventKey = null, $options = array()) {
 95:         parent::attach($callable, $eventKey, $options);

104:  * @see CakeEventManager::detach()

107:     public function detach($callable, $eventKey = null) {
108:         if (is_object($callable)) {

111:         }
112:         parent::detach($callable, $eventKey);
113:     }

115: /**
116:  * Detach all listener objects belonging to a plugin
117:  * @param $plugin string
118:  * @return void
119:  */
120:     public function detachPluginSubscribers($plugin) {
121:         $eventHandlers = Configure::read('EventHandlers');
122:         if (empty($eventHandlers)) {
123:             return;
124:         }
125:         $eventHandlers = array_keys($eventHandlers);
126:         $eventHandlers = preg_grep('/^' . $plugin . '/', $eventHandlers);
127:         foreach ($eventHandlers as $eventHandler) {
128:             list(, $class) = pluginSplit($eventHandler);
129:             if (isset($this->_listenersMap[$class])) {
130:                 $this->detach($this->_listenersMap[$class]);
131:             }
132:         }
133:     }
