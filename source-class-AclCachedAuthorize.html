 28:     protected function _setPrefixMappings() {
 29:         if (!Configure::read('Access Control.rowLevel')) {
 30:             return;
 31:         }
 32: 
 33:         $crud = array('create', 'read', 'update', 'delete');
 34:         $map = array_combine($crud, $crud);
 35: 
 36:         $Controller = $this->Controller();
 37:         if ($Controller->Components->attached('RowLevelAcl')) {
 38:             $settings = $Controller->Components->RowLevelAcl->settings;
 39:             if (isset($settings['actionMap'])) {
 40:                 $map = array_merge($map, $settings['actionMap']);
 41:             }
 42:         }

 44:         $prefixes = Router::prefixes();
 45:         if (!empty($prefixes)) {

 47:                 $map = array_merge($map, array(
 48:                     $prefix . '_moveup' => 'update',
 49:                     $prefix . '_movedown' => 'update',
 50:                     $prefix . '_process' => 'delete',
 51:                     $prefix . '_index' => 'read',
 52:                     $prefix . '_add' => 'create',
 53:                     $prefix . '_edit' => 'update',
 54:                     $prefix . '_view' => 'read',
 55:                     $prefix . '_remove' => 'delete',
 56:                     $prefix . '_create' => 'create',
 57:                     $prefix . '_read' => 'read',
 58:                     $prefix . '_update' => 'update',
 59:                     $prefix . '_delete' => 'delete'
 60:                 ));
 61:             }

 70:     public function authorize($user, CakeRequest $request) {
 71:         $allowed = false;
 72:         $Acl = $this->_Collection->load('Acl');
 73:         list($plugin, $userModel) = pluginSplit($this->settings['userModel']);
 74:         $user = array($userModel => $user);
 75:         $action = $this->action($request);
 76: 
 77:         $cacheName = 'permissions_' . strval($user['User']['id']);
 78:         if (($permissions = Cache::read($cacheName, 'permissions')) === false) {
 79:             $permissions = array();
 80:             Cache::write($cacheName, $permissions, 'permissions');
 81:         }
 82: 
 83:         if (!isset($permissions[$action])) {
 84:             $User = ClassRegistry::init($this->settings['userModel']);
 85:             $User->id = $user['User']['id'];
 86:             $allowed = $Acl->check($User, $action);
 87:             $permissions[$action] = $allowed;
 88:             Cache::write($cacheName, $permissions, 'permissions');
 89:             $hit = false;
 90:         } else {
 91:             $allowed = $permissions[$action];
 92:             $hit = true;
 93:         }
 94: 
 95:         if (Configure::read('debug')) {
 96:             $status = $allowed ? ' allowed.' : ' denied.';
 97:             $cached = $hit ? ' (cache hit)' : ' (cache miss)';
 98:             CakeLog::write(LOG_ERR, $user['User']['username'] . ' - ' . $action . $status . $cached);
 99:         }
100: 
101:         if (!$allowed) {
102:             return false;
103:         }
104: 
105:         if (!Configure::read('Access Control.rowLevel')) {
106:             return $allowed;
107:         }
108: 
109:         // bail out when controller's primary model does not want row level acl
110:         $controller = $this->controller();
111:         $model = $controller->modelClass;
112:         $Model = $controller->{$model};
113:         if ($Model && !$Model->Behaviors->attached('RowLevelAcl')) {
114:             return $allowed;
115:         }
116: 
117:         $primaryKey = $Model->primaryKey;
118:         $ids = array();
119:         if ($request->is('get') && !empty($request->params['pass'][0])) {
120:             // collect id from actions such as: Nodes/admin_edit/1
121:             $ids[] = $request->params['pass'][0];
122:         } elseif ($request->is('post') || $request->is('put') && isset($request->data[$model]['action'])) {
123:             // collect ids from 'bulk' processing action such as: Nodes/admin_process
124:             foreach ($request->data[$model] as $id => $flag) {
125:                 if (isset($flag[$primaryKey]) && $flag[$primaryKey] == 1) {
126:                     $ids[] = $id;
127:                 }
128:             }
129:         }
130:         foreach ($ids as $id) {
131:             if (is_numeric($id)) {
132:                 try {
133:                     $allowed = $this->_authorizeByContent($user['User'], $request, $id);
134:                 } catch (CakeException $e) {
135:                     $allowed = false;
136:                 }
137:             } else {
138:                 continue;
139:             }
140:             if (!$allowed) {
141:                 break;
142:             }
143:         }
144: 
145:         return $allowed;
146:     }

153:     protected function _authorizeByContent($user, CakeRequest $request, $id) {
154:         if (!isset($this->settings['actionMap'][$request->params['action']])) {
155:             throw new CakeException(
156:                 __d('croogo', '_authorizeByContent() - Access of un-mapped action "%1$s" in controller "%2$s"',
157:                 $request->action,
158:                 $request->controller
159:             ));
160:         }
161: 
162:         list($plugin, $userModel) = pluginSplit($this->settings['userModel']);
163:         $user = array($userModel => $user);
164:         $acoNode = array(
165:             'model' => $this->_Controller->modelClass,
166:             'foreign_key' => $id,
167:         );
168:         $alias = sprintf('%s.%s', $acoNode['model'], $acoNode['foreign_key']);
169:         $action = $this->settings['actionMap'][$request->params['action']];
170: 
171:         $cacheName = 'permissions_content_' . strval($user['User']['id']);
172:         if (($permissions = Cache::read($cacheName, 'permissions')) === false) {
173:             $permissions = array();
174:             Cache::write($cacheName, $permissions, 'permissions');
175:         }
176: 
177:         if (!isset($permissions[$alias][$action])) {
178:             $Acl = $this->_Collection->load('Acl');
179:             try {
180:                 $allowed = $Acl->check($user, $acoNode, $action);
181:             } catch (Exception $e) {
182:                 CakeLog::warning('authorizeByContent: ' . $e->getMessage());
183:                 $allowed = false;
184:             }
185:             $permissions[$alias][$action] = $allowed;
186:             Cache::write($cacheName, $permissions, 'permissions');
187:             $hit = false;
188:         } else {
189:             $allowed = $permissions[$alias][$action];
190:             $hit = true;
191:         }
192:

194:             $status = $allowed ? ' allowed.' : ' denied.';
195:             $cached = $hit ? ' (cache hit)' : ' (cache miss)';
196:             CakeLog::write(LOG_ERR, $user['User']['username'] . ' - ' . $action . '/' . $id . $status . $cached);
197:         }
198:         return $allowed;
199:     }
