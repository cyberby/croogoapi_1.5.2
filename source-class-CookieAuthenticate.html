30:  * Authenticates the identity contained in the cookie.  Will use the `settings.userModel`, and `settings.fields`
31:  * to find COOKIE data that is used to find a matching record in the `settings.userModel`.  Will return false if
32:  * there is no cookie data, either username or password is missing, of if the scope conditions have not been met.
33:  *
34:  * @param CakeRequest $request The unused request object
35:  * @param CakeResponse $response Unused response object.
36:  * @return mixed.  False on login failure.  An array of User data on success.
37:  * @throws CakeException
38:  */
39:     public function authenticate(CakeRequest $request, CakeResponse $response) {
40:         if (!isset($this->_Collection->Cookie) || !$this->_Collection->Cookie instanceof CookieComponent) {
41:             throw new CakeException('CookieComponent is not loaded');
42:         }
43: 
44:         $this->settings = array_merge(array('crypt' => 'rijndael'), $this->settings);
45:         if ($this->settings['crypt'] == 'rijndael' && !function_exists('mcrypt_encrypt')) {
46:             throw new CakeException('Cannot use type rijndael, mcrypt_encrypt() is required');
47:         }
48:         $this->_Collection->Cookie->type($this->settings['crypt']);
49: 
50:         list(, $model) = pluginSplit($this->settings['userModel']);
51: 
52:         $data = $this->_Collection->Cookie->read($model);
53:         if (empty($data)) {
54:             return false;
55:         }
56:

58:         if (empty($data[$username]) || empty($data[$password])) {
59:             return false;
