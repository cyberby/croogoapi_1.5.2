181: /**
182:  * Run migration
183:  *
184:  * @param string $direction, up or down direction of migration process
185:  * @return boolean Status of the process
186:  * @throws MigrationException
187:  */
188:     public function run($direction) {
189:         if (!in_array($direction, array('up', 'down'))) {
190:             throw new MigrationException($this, sprintf(
191:                 __d('migrations', 'Migration direction (%s) is not one of valid directions.'), $direction
192:             ), E_USER_NOTICE);
193:         }
194:         $this->direction = $direction;
195: 
196:         $null = null;
197:         $this->db = ConnectionManager::getDataSource($this->connection);
198:         $this->db->cacheSources = false;
199:         $this->db->begin($null);
200:         $this->Schema = new CakeSchema(array('connection' => $this->connection));
201: 
202:         try {
203:             $this->_invokeCallbacks('beforeMigration', $direction);
204:             $result = $this->_run();
205:             $this->_clearCache();
206:             $this->_invokeCallbacks('afterMigration', $direction);
207: 
208:             if (!$result) {
209:                 return false;
210:             }
211:         } catch (Exception $e) {
212:             $this->db->rollback($null);
213:             throw $e;
214:         }
215: 
216:         return $this->db->commit($null);
217:     }

219: /**
220:  * Run migration commands
221:  *
222:  * @return void
223:  * @throws MigrationException
224:  */
225:     protected function _run() {
226:         $result = true;
227:         //force the order of migration types
228:         uksort($this->migration[$this->direction], array($this, 'migration_order'));
229:         foreach ($this->migration[$this->direction] as $type => $info) {
230:             switch ($type) {
231:                 case 'create_table':
232:                     $methodName = '_createTable';
233:                     break;
234:                 case 'drop_table':
235:                     $methodName = '_dropTable';
236:                     break;
237:                 case 'rename_table':
238:                     $methodName = '_renameTable';
239:                     break;
240:                 case 'create_field':
241:                     $type = 'add';
242:                     $methodName = '_alterTable';
243:                     break;
244:                 case 'drop_field':
245:                     $type = 'drop';
246:                     $methodName = '_alterTable';
247:                     break;
248:                 case 'alter_field':
249:                     $type = 'change';
250:                     $methodName = '_alterTable';
251:                     break;
252:                 case 'rename_field':
253:                     $type = 'rename';
254:                     $methodName = '_alterTable';
255:                     break;
256:                 default:
257:                     $message = __d('migrations', 'Migration action type (%s) is not one of valid actions type.', $type);
258:                     throw new MigrationException($this, $message, E_USER_NOTICE);
259:             }
260: 
261:             try {
262:                 $result = $this->{$methodName}($type, $info);
263:             } catch (Exception $e) {
264:                 throw new MigrationException($this, sprintf(__d('migrations', '%s'), $e->getMessage()));
265:             }
266:         }
267:         return $result;
268:     }

352: /**
353:  * Alter Table method
354:  *
355:  * @param string $type Type of operation to be done
356:  * @param array $tables List of tables and fields
357:  * @return boolean Return true in case of success, otherwise false
358:  * @throws MigrationException
359:  */
360:     protected function _alterTable($type, $tables) {
361:         foreach ($tables as $table => $fields) {
362:             $indexes = array();
363:             if (isset($fields['indexes'])) {
364:                 $indexes = $fields['indexes'];
365:                 unset($fields['indexes']);
366:             }
367: 
368:             if ($type == 'drop') {
369:                 $this->_alterIndexes($indexes, $type, $table);
370:             }
371: 
372:             foreach ($fields as $field => $col) {
373:                 $model = new Model(array('table' => $table, 'ds' => $this->connection));
374:                 $tableFields = $this->db->describe($model);
375: 
376:                 if ($type === 'drop') {
377:                     $field = $col;
378:                 }
379: 
380:                 if ($type == 'rename') {
381:                     $data = array('table' => $table, 'old_name' => $field, 'new_name' => $col);
382:                 } else {
383:                     $data = array('table' => $table, 'field' => $field);
384:                 }
385:                 $callbackData = $data;
386:                 if ($this->_invokePrecheck('beforeAction', $type . '_field', $data)) {
387:                     switch ($type) {
388:                         case 'add':
389:                             $sql = $this->db->alterSchema(array(
390:                                 $table => array('add' => array($field => $col))
391:                             ));
392:                             break;
393:                         case 'drop':
394:                             $sql = $this->db->alterSchema(array(
395:                                 $table => array('drop' => array($field => array()))
396:                             ));
397:                             break;
398:                         case 'change':
399:                             if (!isset($col['type']) || $col['type'] == $tableFields[$field]['type']) {
400:                                 $def = array_merge($tableFields[$field], $col);
401:                             } else {
402:                                 $def = $col;
403:                             }
404:                             if (!empty($def['length']) && !empty($col['type']) && (substr($col['type'], 0, 4) == 'date' || substr($col['type'], 0, 4) == 'time')) {
405:                                 $def['length'] = null;
406:                             }
407:                             $sql = $this->db->alterSchema(array(
408:                                 $table => array('change' => array($field => $def))
409:                             ));
410:                             break;
411:                         case 'rename':
412:                             $data = array();
413:                             if (array_key_exists($field, $tableFields)) {
414:                                 $data = $tableFields[$field];
415:                             }
416:                             $sql = $this->db->alterSchema(array(
417:                                 $table => array('change' => array($field => array_merge($data, array('name' => $col))))
418:                             ));
419:                             break;
420:                     }
421: 
422:                     $this->_invokeCallbacks('beforeAction', $type . '_field', $callbackData);
423:                     if (@$this->db->execute($sql) === false) {
424:                         throw new MigrationException($this, sprintf(__d('migrations', 'SQL Error: %s'), $this->db->error));
425:                     }
426:                     $this->_invokeCallbacks('afterAction', $type . '_field', $callbackData);
427:                 }
428:             }
429: 
430:             if ($type != 'drop') {
431:                 $this->_alterIndexes($indexes, $type, $table);
432:             }
433:         }
434:         return true;
435:     }

466: /**
467:  * This method will invoke the before/afterAction callbacks, it is good when
468:  * you need track every action.
469:  *
470:  * @param string $callback Callback name, beforeMigration, beforeAction, afterAction
471:  *      or afterMigration.
472:  * @param string $type Type of action. i.e: create_table, drop_table, etc.
473:  *      Or also can be the direction, for before and after Migration callbacks
474:  * @param array $data Data to send to the callback
475:  * @return void
476:  * @throws MigrationException
477:  */
478:     protected function _invokeCallbacks($callback, $type, $data = array()) {
479:         if ($this->callback !== null && method_exists($this->callback, $callback)) {
480:             if ($callback == 'beforeMigration' || $callback == 'afterMigration') {
481:                 $this->callback->{$callback}($this, $type);
482:             } else {
483:                 $this->callback->{$callback}($this, $type, $data);
484:             }
485:         }
486:         if ($callback == 'beforeMigration' || $callback == 'afterMigration') {
487:             $callback = str_replace('Migration', '', $callback);
488:             if ($this->{$callback}($type)) {
489:                 return;
490:             }
491: 
492:             throw new MigrationException($this, sprintf(
493:                 __d('migrations', 'Interrupted when running "%s" callback.'), $callback
494:             ), E_USER_NOTICE);
495:         }
496:     }

498: /**
499:  * This method will invoke the before/afterAction callbacks, it is good when
500:  * you need track every action.
501:  *
502:  * @param string $callback Callback name, beforeMigration, beforeAction
503:   *         or afterMigration.
504:  * @param string $type Type of action. i.e: create_table, drop_table, etc.
505:  *         Or also can be the direction, for before and after Migration callbacks
506:  * @param array $data Data to send to the callback

509:     protected function _invokePrecheck($callback, $type, $data = array()) {
510:         if ($callback == 'beforeAction') {
511:             return $this->Precheck->{$callback}($this, $type, $data);
512:         }
513:         return false;
