  3: /**
  4:  * RoleLevelAcl Component
  5:  *
  6:  * When "Access Control.rowLevel" Setting is active, this component will perform
  7:  * the necesary setup on controller's primary model and hook the element for 
  8:  * backend use.
  9:  *
 10:  * You can also use it to configure the action mappings used by AclCachedAuthorize
 11:  * class, for example:
 12:  *
 13:  * ```
 14:  *      class ItemsController extends AppController {
 15:  *          public $components = array(
 16:  *              'RowLevelAcl' => array(
 17:  *                  'className' => 'Acl.RowLevelAcl',
 18:  *                  'settings' => array(
 19:  *                      'actionMap' => array(
 20:  *                          'admin_reserve' => 'update', // action map
 21:  *                      ),
 22:  *                  ),
 23:  *              ));
 24:  *      }
 25:  * ```
 26:  *
 27:  * PHP version 5
 28:  *
 29:  * @category Component
 30:  * @package  Croogo.Acl.Controller.Component
 31:  * @version  1.0
 32:  * @author   Fahad Ibnay Heylaal <contact@fahad19.com>
 33:  * @license  http://www.opensource.org/licenses/mit-license.php The MIT License
 34:  * @link     http://www.croogo.org
 35:  */
 36: class RowLevelAclComponent extends Component {
 37: 
 38: /**
 39:  * controller instance
 40:  */
 41:     protected $_controller;
 42: 
 43: /**
 44:  * initialize
 45:  *
 46:  * attaches Acl and RowLevelAcl behavior to the controller's primary model and
 47:  * hook the appropriate admin tabs
 48:  */
 49:     public function initialize(Controller $controller) {
 50:         $this->_controller = $controller;
 51:         $Model = $controller->{$controller->modelClass};
 52:         $Model->Behaviors->load('Acl', array(
 53:             'className' => 'Croogo.CroogoAcl', 'type' => 'controlled',
 54:         ));
 55:         $Model->Behaviors->load('RowLevelAcl', array(
 56:             'className' => 'Acl.RowLevelAcl'
 57:         ));
 58: 
 59:         $name = $controller->name;
 60:         $element = 'Acl.admin/row_acl';
 61:         if (!empty($this->settings['adminTabElement'])) {
 62:             $element = $this->settings['adminTabElement'];
 63:         }
 64:         $adminTabActions = array('admin_add', 'admin_edit');
 65:         if (!empty($this->settings['adminTabActions'])) {
 66:             $adminTabActions += $this->settings['adminTabActions'];
 67:         }
 68:         foreach ($adminTabActions as $action) {
 69:             Croogo::hookAdminTab("$name/$action", __d('croogo', 'Permissions'), $element);
 70:         }
 71:     }
 72: 
 73: /**
 74:  * startup
 75:  */
 76:     public function startup(Controller $controller) {
 77:         if (!empty($controller->request->params['pass'][0])) {
 78:             $id = $controller->request->params['pass'][0];
 79:             $this->_rolePermissions($id);
 80:         }
 81:     }
 82: 
 83: /**
 84:  * Retrieve a list of roles with access status for a given node
 85:  *
 86:  * @param integer $id Node id
 87:  * @return array 
 88:  */
 89:     protected function _rolePermissions($id) {
 90:         $Permission = $this->_controller->Acl->Aro->Permission;
 91:         $Role = ClassRegistry::init('Users.Role');
 92:         $roles = $Role->find('list', array(
 93:             'cache' => array('name' => 'roles', 'config' => 'permissions'),
 94:         ));
 95:         $modelClass = $this->_controller->modelClass;
 96:         $aco = array('model' => $modelClass, 'foreign_key' => $id);
 97:         foreach ($roles as $roleId => $role) {
 98:             $aro = array('model' => 'Role', 'foreign_key' => $roleId);
 99:             try {
100:                 $allowed = $Permission->check($aro, $aco);
101:             } catch (CakeException $e) {
102:                 $allowed = false;
103:             }
104:             $rolePermissions[] = array(
105:                 'Role' => array(
106:                     'id' => $roleId, 'title' => $role, 'allowed' => $allowed
107:                 )
108:             );
109:         }
110:         $this->_controller->set(compact('rolePermissions'));
111:     }
112: 
113: }
